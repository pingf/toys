首先来看一下ST公司针对STM32F4系列器件给出的驱动库的目录结构

Libraries目录下为实际使用的驱动库,其又分为CMSIS和STM32F4xx_StdPeriph_Driver两个目录,前者为引导相关文件,后者为实际驱动库.相较于STM32F1系列,CMSIS目录下多了不少东西,比如增加了和RTOS和DSP指令集相关的东西

Project下面是例子和模板

Utilities下面是一些和具体板子相关的工程




然后在eclipse中通过gnuarm4eclipse插件新建个工程,并拷贝下图中的文件至对应的目录

STM32F4xx_StdPeriph_Driver目录就是从库中照搬过来滴

core目录放了些杂七杂八的东西,要关注这么几个

那个.S文件是汇编引导文件,.ld文件是连接器链接文件,.stlink是gdb用来调用调用stlink下载和调试用的文件,.cfg是openocd用来配置stlink的文件,.stm32f4xx_conf.h是配置文件[需要用哪一部分的库,要在这个文件中通过注释来选择],stm32f4xx.h是CMSIS层总体的头文件,相当多的宏定义都在这里.system_stm32f4xx.c和.h是一些系统初始化函数,比如systeminit函数之类.stm32f4xx_it.c和.h是一些定义好的中断相应函数[不用的时候可以不让其参与编译].而以core打头的则是一些特别的函数,不少内嵌了汇编.

这里另外的proj目录用来放自己写的程序好了

那个Debug目录则是eclipse自己生成的,会debug版本的代码




这里另外的proj目录用来放自己写的程序好了,下面的是一个简单的例子[放在proj目录下]

#include "stm32f4xx_conf.h"
void delay(unsigned int cnt)
{
  while(cnt--) ;
}
int main()
{
	  GPIO_InitTypeDef  GPIO_InitStructure;
	  /* GPIOD Periph clock enable */
	  RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOD, ENABLE); 
	  /* Configure PD12 and PD13 in output pushpull mode */
	  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_12 | GPIO_Pin_13;
	  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
	  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
	  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz;
	  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL;
	  GPIO_Init(GPIOD, &GPIO_InitStructure); 
	  while (1)
	  {
	    /* Set PD12 and PD13 */
	    GPIOD->BSRRL = GPIO_Pin_12 | GPIO_Pin_13;
	    delay(5000);
	    /* Reset PD12 and PD13 */
	    GPIOD->BSRRH = GPIO_Pin_12 | GPIO_Pin_13;
	    delay(5000); 
	  }
  return 0;
}

首先配置环境变量

如图,要将gcc的toolchain的bin目录加入Path中;另外将头文件所在的目录加入C_INCLUDE_PATH中

加入宏定义

STM32F40XX是用来选择芯片型号的宏,407系列可以用这个;USE_STDPERIPH_DRIVER是指明使用驱动库的宏,如果不用可能会报一些跟cs-make出错之类的错误,窝也不知为啥会报这样的错,还以为是啥未定义之类的.不过只要这两个宏定义了就欧科了

设置linkerscript文件

这个文件是链接器用的,可以从库中找到的,不过下面还是罗列一下好了

/* Entry Point */
ENTRY(Reset_Handler)

/* Highest address of the user mode stack */
_estack = 0x20020000;    /* end of 128K RAM on AHB bus*/

/* Generate a link error if heap and stack don't fit into RAM */
_Min_Heap_Size = 0;      /* required amount of heap  */
_Min_Stack_Size = 0x400; /* required amount of stack */

/* Specify the memory areas */
MEMORY
{
  FLASH (rx)      : ORIGIN = 0x08000000, LENGTH = 1M
  RAM (xrw)       : ORIGIN = 0x20000000, LENGTH = 128K
  CCMRAM (xrw)    : ORIGIN = 0x10000000, LENGTH = 64K
  MEMORY_B1 (rx)  : ORIGIN = 0x60000000, LENGTH = 0K
}

/* Define output sections */
SECTIONS
{
  /* The startup code goes first into FLASH */
  .isr_vector :
  {
    . = ALIGN(4);
    KEEP(*(.isr_vector)) /* Startup code */
    . = ALIGN(4);
  } >FLASH

  /* The program code and other data goes into FLASH */
  .text :
  {
    . = ALIGN(4);
    *(.text)           /* .text sections (code) */
    *(.text*)          /* .text* sections (code) */
    *(.rodata)         /* .rodata sections (constants, strings, etc.) */
    *(.rodata*)        /* .rodata* sections (constants, strings, etc.) */
    *(.glue_7)         /* glue arm to thumb code */
    *(.glue_7t)        /* glue thumb to arm code */
	*(.eh_frame)

    KEEP (*(.init))
    KEEP (*(.fini))

    . = ALIGN(4);
    _etext = .;        /* define a global symbols at end of code */
  } >FLASH


   .ARM.extab   : { *(.ARM.extab* .gnu.linkonce.armextab.*) } >FLASH
    .ARM : {
    __exidx_start = .;
      *(.ARM.exidx*)
      __exidx_end = .;
    } >FLASH

  .preinit_array     :
  {
    PROVIDE_HIDDEN (__preinit_array_start = .);
    KEEP (*(.preinit_array*))
    PROVIDE_HIDDEN (__preinit_array_end = .);
  } >FLASH
  .init_array :
  {
    PROVIDE_HIDDEN (__init_array_start = .);
    KEEP (*(SORT(.init_array.*)))
    KEEP (*(.init_array*))
    PROVIDE_HIDDEN (__init_array_end = .);
  } >FLASH
  .fini_array :
  {
    PROVIDE_HIDDEN (__fini_array_start = .);
    KEEP (*(.fini_array*))
    KEEP (*(SORT(.fini_array.*)))
    PROVIDE_HIDDEN (__fini_array_end = .);
  } >FLASH

  /* used by the startup to initialize data */
  _sidata = .;

  /* Initialized data sections goes into RAM, load LMA copy after code */
  .data : AT ( _sidata )
  {
    . = ALIGN(4);
    _sdata = .;        /* create a global symbol at data start */
    *(.data)           /* .data sections */
    *(.data*)          /* .data* sections */

    . = ALIGN(4);
    _edata = .;        /* define a global symbol at data end */
  } >RAM

  /* Uninitialized data section */
  . = ALIGN(4);
  .bss :
  {
    /* This is used by the startup in order to initialize the .bss secion */
    _sbss = .;         /* define a global symbol at bss start */
    __bss_start__ = _sbss;
    *(.bss)
    *(.bss*)
    *(COMMON)

    . = ALIGN(4);
    _ebss = .;         /* define a global symbol at bss end */
    __bss_end__ = _ebss;
  } >RAM

  /* User_heap_stack section, used to check that there is enough RAM left */
  ._user_heap_stack :
  {
    . = ALIGN(4);
    PROVIDE ( end = . );
    PROVIDE ( _end = . );
    . = . + _Min_Heap_Size;
    . = . + _Min_Stack_Size;
    . = ALIGN(4);
  } >RAM

  /* MEMORY_bank1 section, code must be located here explicitly            */
  /* Example: extern int foo(void) __attribute__ ((section (".mb1text"))); */
  .memory_b1_text :
  {
    *(.mb1text)        /* .mb1text sections (code) */
    *(.mb1text*)       /* .mb1text* sections (code)  */
    *(.mb1rodata)      /* read-only data (constants) */
    *(.mb1rodata*)
  } >MEMORY_B1

  /* Remove information from the standard libraries */
  /DISCARD/ :
  {
    libc.a ( * )
    libm.a ( * )
    libgcc.a ( * )
  }

  .ARM.attributes 0 : { *(.ARM.attributes) }
}

这个配置会把程序下载到flash中,如果有需要可以修改成下载到ram中,延长使用寿命

然后点击Build编译就好了
