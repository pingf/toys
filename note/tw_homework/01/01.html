import static org.junit.Assert.assertEquals;
import org.junit.Before;
import org.junit.Test;

public class TestConverter {
    Converter converter;
    String ret;
    @Before
    public void setup() {
        converter = new Converter();
    }
    @Test
    public void hmm_should_not_get_mommified() {
        ret = converter.convert("hmm");
        assertEquals("hmm", ret);
    }
    @Test
    public void ship_should_not_get_mommified() {
        ret = converter.convert("ship");
        assertEquals("ship", ret);
    }
    @Test
    public void his_should_get_mommified() {
        ret = converter.convert("his");
        assertEquals("hmommys",ret);
    }
    @Test
    public void she_should_get_mommified() {
        ret = converter.convert("she");
        assertEquals("shmommy",ret);
    }
    @Test
    public void hear_should_get_mommified() {
        ret = converter.convert("hear");
        assertEquals("hmommyr",ret);
    }
    @Test
    public void shape_should_get_mommified() {
        ret = converter.convert("shape");
        assertEquals("shmommypmommy",ret);
    }
    @Test
    public void banana_should_get_mommified() {
        ret = converter.convert("banana");
        assertEquals("bmommynmommynmommy",ret);
    }
}



public class Converter{
    private static final String vowels = "aeiou";
    private static final String pattern = "(a|e|i|o|u)+"; 
    private double vowels_percent(String s) {
        int cnt=0;
        for (char c : s.toCharArray()) {
            if (vowels.indexOf(c)>-1) {
                cnt++;
            }
        }
        return cnt/(double)s.length();
    }
    public String convert(String s) {
        if (vowels_percent(s) > 0.3) {
            s=s.replaceAll(pattern,"mommy");
        } 
        return s; 
    }
}
